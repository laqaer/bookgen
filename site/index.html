Dead simple to use

# Install
git clone https://github.com/laqaer/bookgen.git
cd bookgen

# Generate a complete book
python3 bookgen.py books/agent_economics.yaml

# Or use AI for higher quality
python3 bookgen.py books/agent_economics.yaml --llm

# Full pipeline: manuscript + marketing + everything
python3 bookops.py generate-all books/agent_economics.yaml
